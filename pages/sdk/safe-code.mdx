### Getting current version

```tsx
import { Vault } from 'bakosafe'

const currentVersion = await Vault.BakoSafeGetCurrentVersion()
```

### Getting a specific version

```tsx
import { Vault } from 'bakosafe'

// safe_code is a unique string that identifies your predicate contract
const safe_code = 'real_code'

const version = await Vault.BakoSafeGetVersionByCode(safe_code)
```

### Listing versions

```tsx
import { Vault } from 'bakosafe'

const versions = await Vault.BakoSafeGetVersions()
```
